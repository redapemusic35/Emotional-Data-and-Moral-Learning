```r Country, echo = FALSE} library(gridExtra) data(mtcars) library(ggplot2) setwd(“~/mallet/”) keys <- read.csv(“Rap_keys1.csv”, header = TRUE, sep = “,”) Edges <- read.csv(“Edges.csv”, header = TRUE, sep = “,”) all <- merge(keys, Edges, by = “Topics”)

ggplot(all, aes(Topics, Order)) + geom_point(color = “red”) + geom_text_repel(data = subset(all, Order > 0.04), segment.size = 0.15, segment.alpha = 1.0, aes( color = factor(Topics), label = Words, )) ```
